from braket.circuits import Circuit

circuit = Circuit()

# Apply optimizations
# Original gates: X(0), X(0), CNOT(0, 1), CNOT(0, 1), H(0), H(0)
# Z(1), Z(1), X(1), X(1)

# Simplify redundant gates
# Two consecutive X gates on the same qubit cancel each other out
# Two consecutive Z gates on the same qubit cancel each other out

# Optimized circuit
circuit.cnot(0, 1)
circuit.cnot(0, 1)
circuit.h(0)
circuit.h(0)
circuit.x(1)

# Print the optimized circuit
print(circuit)
